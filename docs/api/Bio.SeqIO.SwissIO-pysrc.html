 30      An offset of -1 is used with a start location to make it pythonic. 
 31      """ 
 32      if location_string == "?": 
 33          return SeqFeature.UnknownPosition() 
 34      # Hack so that feature from 0 to 0 becomes 0 to 0, not -1 to 0. 
 35      try: 
 36          return SeqFeature.ExactPosition(max(0, offset + int(location_string))) 
 37      except ValueError:

 39      if location_string.startswith("<"): 
 40          try: 
 41              return SeqFeature.BeforePosition(max(0, offset + int(location_string[1:]))) 
 42          except ValueError: 
 43              pass

 45          try: 
 46              return SeqFeature.AfterPosition(max(0, offset + int(location_string[1:])))

 48              pass 
 49      elif location_string.startswith("?"):  # e.g. "?22"
