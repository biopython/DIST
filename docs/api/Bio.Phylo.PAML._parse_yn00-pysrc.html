 11 -def parse_ng86(lines, results): 

 12      """Parse the Nei & Gojobori (1986) section of the results. 
 13   
 14      Nei_Gojobori results are organized in a lower 
 15      triangular matrix, with the sequence names labeling 
 16      the rows and statistics in the format: 
 17      w (dN dS) per column 
 18      Example row (2 columns): 
 19      0.0000 (0.0000 0.0207) 0.0000 (0.0000 0.0421) 
 20      """ 
 21      sequences = [] 
 22      for line in lines: 
 23          # The purpose of this complex regex is to parse the NG86 section for 
 24          # valid lines of data that are mixed in with citations and comments. 
 25          # The data lines begin with a taxon name, followed by zero or more 
 26          # fields containing numeric values, sometimes enclosed in parens. 
 27          # Taxon names are from 1-30 characters and are usually separated from 
 28          # the numeric portion of the line by space(s).  Long taxon names to are 
 29          # truncated to 30 characters, and may run into the data fields without 
 30          # any separator., e.g. some_long_name-1.0000 
 31          # This regex is an attempt to cover more pathological cases while also 
 32          # parsing all existing versions of yn00 output with shorter names. 
 33   
 34          matrix_row_res = re.match(r"^([^\s]+?)(\s+-?\d+\.\d+.*$|\s*$|-1.0000\s*\(.*$)", line) 
 35          if matrix_row_res is not None: 
 36              # Find all floating point numbers in this line, accounting 
 37              # for the fact that the sequence IDs might have bits that 
 38              # look like floating point values. 
 39              line_floats_res = re.findall(r"-*\d+\.\d+", matrix_row_res.group(2)) 
 40              line_floats = [float(val) for val in line_floats_res] 
 41   
 42              seq_name = matrix_row_res.group(1).strip() 
 43              sequences.append(seq_name) 
 44              results[seq_name] = {} 
 45              for i in range(0, len(line_floats), 3): 
 46                  NG86 = {} 
 47                  NG86["omega"] = line_floats[i] 
 48                  NG86["dN"] = line_floats[i + 1] 
 49                  NG86["dS"] = line_floats[i + 2] 
 50                  results[seq_name][sequences[i // 3]] = {"NG86": NG86} 
 51                  results[sequences[i // 3]][seq_name] = {"NG86": NG86} 
 52      return (results, sequences)

 66      # +- 0.0151 
 67      for line in lines: 
 68          # Find all floating point numbers in this line 
 69          line_floats_res = re.findall(r"-*\d+\.\d+", line) 
 70          line_floats = [float(val) for val in line_floats_res] 
 71          row_res = re.match(r"\s+(\d+)\s+(\d+)", line) 
 72          if row_res is not None: 
 73              seq1 = int(row_res.group(1)) 
 74              seq2 = int(row_res.group(2)) 
 75              seq_name1 = sequences[seq1 - 1] 
 76              seq_name2 = sequences[seq2 - 1] 
 77              YN00 = {} 
 78              YN00["S"] = line_floats[0] 
 79              YN00["N"] = line_floats[1] 
 80              YN00["t"] = line_floats[2] 
 81              YN00["kappa"] = line_floats[3]

 83              YN00["dN"] = line_floats[5] 
 84              YN00["dN SE"] = line_floats[6]
