 70      """Parse a GCK file and return a SeqRecord object.

 72      Note that a GCK file can only contain one sequence, so this 
 73      iterator will always return a single record. 
 74      """ 
 75      # Skip file header 
 76      # GCK files start with a 24-bytes header. Bytes 4 and 8 seem to 
 77      # always be 12, maybe this could act as a magic cookie. Bytes 
 78      # 17-20 and 21-24 contain variable values of unknown meaning. 
 79      # check if file is empty 
 80      _read_header(handle, 24) 
 81   
 82      # Read the actual sequence data 
 83      packet, length = _read_packet(handle) 
 84      # The body of the sequence packet starts with a 32-bit integer 
 85      # representing the length of the sequence. 
 86      seq_length = unpack(">I", packet[:4])[0] 
 87      # This length should not be larger than the length of the 
 88      # sequence packet. 
 89      if seq_length > length - 4: 
 90          raise ValueError("Conflicting sequence length values") 
 91      sequence = packet[4:].decode("ASCII") 
 92      record = SeqRecord(Seq(sequence, alphabet=Alphabet.generic_dna)) 
 93   
 94      # Skip unknown packet 
 95      _read_packet(handle) 
 96   
 97      # Read features packet 
 98      packet, length = _read_packet(handle) 
 99      (seq_length, num_features) = unpack(">IH", packet[:6]) 
100      # Check that length in the features packet matches the actual 
101      # length of the sequence 
102      if seq_length != len(record): 
103          raise ValueError("Conflicting sequence length values") 
104      # Each feature is stored in a 92-bytes structure. 
105      if length - 6 != num_features * 92: 
106          raise ValueError("Features packet size inconsistent with number of features") 
107      for i in range(0, num_features): 
108          offset = 6 + i * 92 
109          feature_data = packet[offset : offset + 92] 
110   
111          # There's probably more stuff to unpack in that structure, 
112          # but those values are the only ones I understand. 
113          (start, end, type, strand, has_name, has_comment, version) = unpack( 
114              ">II6xH14xB17xII35xB", feature_data 
115          ) 
116   
117          if strand == 1:  # Reverse strand 
118              strand = -1 
119          else: 
120              # Other possible values are 0 (no strand specified), 
121              # 2 (forward strand), and 3 (both strands). All are 
122              # treated as a forward strand. 
123              strand = 1 
124          location = FeatureLocation(start, end, strand=strand) 
125   
126          # It looks like any value > 0 indicates a CDS... 
127          if type > 0: 
128              type = "CDS" 
129          else: 
130              type = "misc_feature" 
131   
132          # Each feature may have a name and a comment, which are then 
133          # stored immediately after the features packet. Names are 
134          # stored as Pascal strings (1 length byte followed by the 
135          # string itself), comments are stored as "32-bit Pascal strings" 
136          # (4 length bytes followed by the string). 
137          qualifiers = {} 
138          if has_name > 0: 
139              name = _read_pstring(handle) 
140              qualifiers["label"] = [name] 
141          if has_comment > 0: 
142              comment = _read_p4string(handle) 
143              qualifiers["note"] = [comment] 
144   
145          # Each feature may exist in several "versions". We keep only 
146          # the most recent version. 
147          if version > 0: 
148              continue 
149   
150          feature = SeqFeature(location, type=type, qualifiers=qualifiers) 
151          record.features.append(feature) 
152   
153      # Read restriction sites packet 
154      # We are not interested in restriction sites, but we must still read 
155      # that packet so that we can skip the names and comments for each 
156      # site, which are stored after that packet in a similar way as for 
157      # the features above. 
158      packet, length = _read_packet(handle) 
159      (seq_length, num_sites) = unpack(">IH", packet[:6]) 
160      # Each site is stored in a 88-bytes structure 
161      if length - 6 != num_sites * 88: 
162          raise ValueError("Sites packet size inconsistent with number of sites") 
163      for i in range(0, num_sites): 
164          offset = 6 + i * 88

166   
167          (start, end, has_name, has_comment) = unpack(">II24xII48x", site_data) 
168   
169          # Skip names and comments 
170          if has_name:

172          if has_comment: 
173              _read_p4string(handle)

176      _read_packet(handle)

178      # Next in the file are "version packets". 
179      # However they are not properly formatted "packets" as they are not 
180      # preceded by an integer giving their size. Instead we have a 
181      # short integer indicating how many versions are there, and then 
182      # as many 260-bytes block as we have versions. 
183      num_versions = _read(handle, 2) 
184      num_versions = unpack(">H", num_versions)[0] 
185      versions = _read(handle, num_versions * 260) 
186      for i in range(0, num_versions):

188          version_data = versions[offset : offset + 260] 
189

191          # after all the "version packets". 
192          has_comment = unpack(">I", version_data[-4:])[0] 
193          if has_comment > 0: 
194              _read_p4string(handle)

198      # indicator, so I hope its size is indeed constant in all files...
