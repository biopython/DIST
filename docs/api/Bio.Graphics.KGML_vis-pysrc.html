 53 -def color_to_reportlab(color): 

 54      """Return the passed color in Reportlab Color format. 
 55   
 56      We allow colors to be specified as hex values, tuples, or Reportlab Color 
 57      objects, and with or without an alpha channel. This function acts as a 
 58      Rosetta stone for conversion of those formats to a Reportlab Color 
 59      object, with alpha value. 
 60   
 61      Any other color specification is returned directly 
 62      """ 
 63      # Reportlab Color objects are in the format we want already 
 64      if isinstance(color, colors.Color): 
 65          return color 
 66      elif isinstance(color, str):  # String implies hex color 
 67          if color.startswith("0x"):  # Standardise to octothorpe 
 68              color.replace("0x", "#") 
 69          if len(color) == 7: 
 70              return colors.HexColor(color) 
 71          else: 
 72              try: 
 73                  return colors.HexColor(color, hasAlpha=True) 
 74              except TypeError:  # Catch pre-2.7 Reportlab 
 75                  raise RuntimeError("Your reportlab seems to be too old, try 2.7 onwards") 
 76      elif isinstance(color, tuple):  # Tuple implies RGB(alpha) tuple 
 77          return colors.Color(*color) 
 78      return color
