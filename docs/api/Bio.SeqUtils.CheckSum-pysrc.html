 79      All sequences are converted to uppercase.

 82          # Assume its a Seq object 
 83          seq = str(seq) 
 84      except AttributeError: 
 85          # Assume its a string

 88      for char in seq: 
 89          index += 1 
 90          checksum += index * ord(char.upper())

111          # Assume it's a Seq object
