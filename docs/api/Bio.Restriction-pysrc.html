 57  #   OK can't put the following code in Bio.Restriction.__init__ unless 
 58  #   I put everything from Restriction in here. 
 59  #   or at least the RestrictionBatch class. 
 60  # 
 61  #   The reason for that is if I do that, I break the __contains__ method of 
 62  #   the RestrictionBatch in Restriction, which expect to find the name of 
 63  #   the enzymes in the locals() dictionary when evaluating string to see if 
 64  #   it is an enzyme. 
 65  # 
 66  #   This call for some explanations I guess: 
 67  #       When testing for the presence of a Restriction enzyme in a 
 68  #       RestrictionBatch, the user can use: 
 69  # 
 70  #           1) a class of type 'RestrictionType' 
 71  #           2) a string of the name of the enzyme (it's repr) 
 72  #               i.e: 
 73  #                   >>> from Bio.Restriction import RestrictionBatch, EcoRI 
 74  #                   >>> MyBatch = RestrictionBatch(EcoRI) 
 75  #                   >>> #!/usr/bin/env python 
 76  #                   >>> EcoRI in MyBatch        # the class EcoRI. 
 77  #                   True 
 78  #                   >>> 
 79  #                   >>> 'EcoRI' in MyBatch      # a string representation 
 80  #                   True 
 81  # 
 82  #   OK, that's how it is suppose to work. And I find it quite useful. 
 83  # 
 84  #   Now if I leave the code here I got: 
 85  #                   >>> from Bio.Restriction import RestrictionBatch, EcoRI 
 86  #                   >>> MyBatch = RestrictionBatch(EcoRI) 
 87  #                   >>> EcoRI in MyBatch # the class EcoRI. 
 88  #                   True 
 89  #                   >>> 'EcoRI' in MyBatch   # a string. 
 90  #                   False 
 91   
 92  #   There is 5 ways to change that: 
 93  #       1) abandon the evaluation of string representation. 
 94  #       2) leave the code like that and hack something in RestrictionBatch. 
 95  #       3) Move back the code in Bio.Restriction.Restriction 
 96  #       4) Move RestrictionBatch here. 
 97  #       5) Remove Restriction.Restriction and move all the code in here 
 98  # 
 99  #   1) no fun in that. 
100  #   2) there is a simpler way to do it. 
101  #   3) I prefer to keep all the code together. 
102  #   4) and 5) both are OK. Only a matter of preference. 
103  # 
104  #   So the following code has been moved back to Bio.Restriction.Restriction 
105  #   For the user the results is transparent: 
106  #   from Bio.Restriction import * works as before.

109  # ## 
110  # ##   The restriction enzyme classes are created dynamically when the module 
111  # ##   is imported. Here is the magic which allow the creation of the 
112  # ##   restriction-enzyme classes. 
113  # ## 
114  # ##   The reason for the two dictionaries in Restriction_Dictionary 
115  # ##   one for the types (which will be called pseudo-type as they really 
116  # ##   correspond to the values that instances of RestrictionType can take) 
117  # ##   and one for the enzymes is efficiency as the bases are evaluated 
118  # ##   once per pseudo-type. 
119  # ## 
120  # ##   However Restriction is still a very inefficient module at import. But 
121  # ##   remember that around 660 classes (which is more or less the size of 
122  # ##   Rebase) have to be created dynamically. However, this processing take 
123  # ##   place only once. 
124  # ##   This inefficiency is however largely compensated by the use of metaclass 
125  # ##   which provide a very efficient layout for the class themselves mostly 
126  # ##   alleviating the need of if/else loops in the class methods. 
127  # ## 
128  # ##   It is essential to run Restriction with doc string optimisation (-OO 
129  # ##   switch) as the doc string of 660 classes take a lot of processing. 
130  # ## 
131  # # CommOnly    = RestrictionBatch()    # commercial enzymes 
132  # # NonComm     = RestrictionBatch()    # not available commercially 
133  # # for TYPE, (bases, enzymes) in typedict.items(): 
134  # #    # 
135  # #    #   The keys are the pseudo-types TYPE (stored as type1, type2...) 
136  # #    #   The names are not important and are only present to differentiate 
137  # #    #   the keys in the dict. All the pseudo-types are in fact 
138  # #    #   RestrictionType. These names will not be used after and the pseudo- 
139  # #    #   types are not kept in the locals() dictionary. It is therefore 
140  # #    #   impossible to import them. 
141  # #    #   Now, if you have look at the dictionary, you will see that not all 
142  # #    #   thet ypes are present as those without corresponding enzymes have 
143  # #    #   been removed by Dictionary_Builder(). 
144  # #    # 
145  # #    #   The values are tuples which contain 
146  # #    #   as first element a tuple of bases (as string) and 
147  # #    #   as second element the names of the enzymes. 
148  # #    # 
149  # #    #   First eval the bases. 
150  # #    # 
151  # #    bases = tuple(eval(x) for x in bases) 
152  # #    # 
153  # #    #   now create the particular value of RestrictionType for the classes 
154  # #    #   in enzymes. 
155  # #    # 
156  # #    T = type.__new__(RestrictionType, 'RestrictionType', bases, {}) 
157  # #    for k in enzymes: 
158  # #        # 
159  # #        #   Now, we go through all the enzymes and assign them their type. 
160  # #        #   enzymedict[k] contains the values of the attributes for this 
161  # #        #   particular class (self.site, self.ovhg,....). 
162  # #        # 
163  # #        newenz = T(k, bases, enzymedict[k]) 
164  # #        # 
165  # #        #   we add the enzymes to the corresponding batch. 
166  # #        # 
167  # #        #   No need to verify the enzyme is a RestrictionType -> add_nocheck 
168  # #        # 
169  # #        if newenz.is_comm() : CommOnly.add_nocheck(newenz) 
170  # #        else : NonComm.add_nocheck(newenz) 
171  # ## 
172  # ##   AllEnzymes is a RestrictionBatch with all the enzymes from Rebase. 
173  # ## 
174  # # AllEnzymes = CommOnly | NonComm 
175  # ## 
176  # ##   Now, place the enzymes in locals so they can be imported. 
177  # ## 
178  # # names = [str(x) for x in AllEnzymes] 
179  # # locals().update(dict(map(None, names, AllEnzymes))) 
180  # ## 
181  # ##   Limit what can be imported by from Restriction import * 
182  # ##   Most of the classes here will never be used outside this module 
183  # ##   (Defined,Palindromic...). It is still possible to request them 
184  # ##   specifically 
185  # ## 
186  # ##   also delete the variable that are no longer needed. 
187  # ## 
188  # ## 
189  # # __all__= ['Analysis', 'RestrictionBatch','AllEnzymes','CommOnly', 
190  # #           'NonComm']+names 
191  # # del k, x, enzymes, TYPE, bases, names
