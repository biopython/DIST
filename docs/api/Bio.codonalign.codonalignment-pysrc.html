312      D = {}  # Final distances dict 
313      P = {}  # Predecessor dict 
314      # Fill the dicts with default values 
315      for node in graph.keys():

320      while len(unseen_nodes) > 0: 
321          # Select the node with the lowest value in D (final distance) 
322          shortest = None 
323          node = ''

325              if shortest is None:

327                  node = temp_node 
328              elif D[temp_node] < shortest:

330                  node = temp_node

332          unseen_nodes.remove(node)
